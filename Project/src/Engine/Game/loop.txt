// The body of the main game loop could be implemented as an EventHandler for GameUpdateEvent which would call the following Update() function:
// Note: The update interval could be specified as a parameter of GameUpdateEvent

void Update() {
	Input();
	Game();
	Render();
	Reset();
}

void Input() {
	static auto model = InputModel::GetInstance();
	static auto event = new InputUpdateEvent( model );
	InputController::HandleEvent( event );
}

void Game() {
	// Update game state...
}

void Render() {
	static auto model = RenderModel::GetInstance();
	static auto event = new RenderDrawEvent( model );
	RenderController::HandleEvent( event );
}

void Reset() {
	static auto model = ClockModel::GetInstance();
	static auto controller = GameController::GetInstance();
	static auto event = new ClockTimerEvent( model, controller );
	ClockController::HandleEvent( event );
}
